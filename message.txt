local player = game.Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local humanoidRootPart = character:WaitForChild("HumanoidRootPart")

local function spawnUnitRepeatedly()
    local unitNames = {"MageChild"}
    local spawnPosition = Vector3.new(-113.44709, 175.723358, 1082.33679)  -- Belirtilen konum

    for _, unitName in pairs(unitNames) do
        local args = {
            [1] = unitName,
            [2] = CFrame.new(spawnPosition),
            [3] = 1,
            [4] = {
                [1] = "1",
                [2] = "1",
                [3] = "1",
                [4] = "1"
            }
        }

        print("Attempting to spawn unit:", unitName)

        local success, errorMessage = pcall(function()
            local remote = game:GetService("ReplicatedStorage"):WaitForChild("Remote"):WaitForChild("SpawnUnit")
            if remote then
                remote:InvokeServer(unpack(args))
            else
                error("Remote not found!")
            end
        end)

        if not success then
            warn("SpawnUnit call failed for " .. unitName .. ": " .. errorMessage)
        else
            print("SpawnUnit call successful for " .. unitName)
        end
    end

    wait(0.5)  -- Bekleme süresi 0.5 saniye olarak güncellendi
    spawnUnitRepeatedly()
end

spawnUnitRepeatedly()
